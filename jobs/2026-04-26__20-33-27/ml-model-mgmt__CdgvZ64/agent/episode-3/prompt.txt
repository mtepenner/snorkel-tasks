from __future__ import annotations

import io
from typing import Any, List, Optional

import numpy as np
import pandas as pd
from flask import Flask, jsonify, request, render_template

app = Flask(__name__, template_folder="templates", static_folder="static")

# In-memory store of the latest processed dataset
_PROCESSED_RECORDS: Optional[List[dict]] = None


def _is_blank(s: Any) -> bool:
    return s is None or (isinstance(s, str) and s.strip() == "")


def _looks_numeric_string(s: Any) -> bool:
    if s is None:
        return False
    if isinstance(s, (int, float, np.integer, np.floating)):
        return True
    if not isinstance(s, str):
        return False
    t = s.strip()
    if t == "":
        return False
    try:
        float(t)
        return True
    except Exception:
        return False


def _reject_headerless_csv(header_cells: List[str]) -> bool:
    """Return True if header row looks like data (e.g., '1,2') or empty."""
    if not header_cells or all(_is_blank(x) for x in header_cells):
        return True

    # If every header cell is numeric/blank, it's almost certainly headerless data.
    numeric_or_blank = [(_looks_numeric_string(x) or _is_blank(x)) for x in header_cells]
    if all(numeric_or_blank):
        return True

    # Also reject if any header cell is an empty string.
    if any(isinstance(x, str) and x.strip() == "" for x in header_cells):
        return True

    return False


def _preprocess(df: pd.DataFrame) -> pd.DataFrame:
    if df is None or df.empty:
        raise ValueError("empty")

    df = df.copy()

    # Identify numeric vs categorical columns.
    numeric_cols = df.select_dtypes(include=["number", "bool"]).columns.tolist()
    cat_cols = [c for c in df.columns if c not in numeric_cols]

    # Numeric: impute mean, then z-score.
    if numeric_cols:
        for c in numeric_cols:
            # coerce bool to float for consistent mean/std
            df[c] = pd.to_numeric(df[c], errors="coerce")
            mean = df[c].mean(skipna=True)
            if pd.isna(mean):
                mean = 0.0
            df[c] = df[c].fillna(mean)
            std = df[c].std(ddof=0)
            if pd.isna(std) or std == 0:
                df[c] = df[c] - df[c].mean()
            else:
                df[c] = (df[c] - df[c].mean()) / std

    # Categorical: fill missing, one-hot encode, drop originals.
    if cat_cols:
        for c in cat_cols:
            df[c] = df[c].astype("string")
            df[c] = df[c].fillna("missing")
        dummies = pd.get_dummies(df[cat_cols], prefix=cat_cols, dummy_na=False)
        df = pd.concat([df.drop(columns=cat_cols), dummies], axis=1)

    # Replace any remaining NaN with None for JSON.
    df = df.replace({np.nan: None})
    return df

@app.post("/api/v1/data/upload")
def upload_data():
    global _PROCESSED_RECORDS

    content_type = (request.content_type or "").split(";")[0].strip().lower()

    try:
        # Shape 1: JSON array of objects
        if content_type == "application/json":
            payload = request.get_json(silent=True)
            if not isinstance(payload, list) or len(payload) == 0:
                return jsonify({"error": "malformed or empty input"}), 400
            if not all(isinstance(x, dict) for x in payload):
                return jsonify({"error": "malformed or empty input"}), 400

            df = pd.DataFrame(payload)

        # Shape 2: raw CSV text with a real header row
        elif content_type in {"text/csv", "text/plain"}:
            raw = request.get_data(as_text=True)
            if raw is None or raw.strip() == "":
                return jsonify({"error": "malformed or empty input"}), 400

            # Check first line for header validity before letting pandas treat it as headers.
            first_line = raw.splitlines()[0] if raw.splitlines() else ""
            header_cells = [c.strip() for c in first_line.split(",")]
            if _reject_headerless_csv(header_cells):
                return jsonify({"error": "CSV must include a header row"}), 400

            # Now parse with pandas; keep_default_na so blanks become NaN.
            try:
                df = pd.read_csv(io.StringIO(raw))
            except Exception:
                return jsonify({"error": "malformed or empty input"}), 400

            if df.empty:
                return jsonify({"error": "malformed or empty input"}), 400

        else:
            return jsonify({"error": "unsupported content type"}), 415

        processed = _preprocess(df)
        _PROCESSED_RECORDS = processed.to_dict(orient="records")
        return jsonify({"status": "ok"}), 200

    except ValueError:
        return jsonify({"error": "malformed or empty input"}), 400
    except Exception:
        # Bad input must never bubble into a 500
        return jsonify({"error": "malformed or empty input"}), 400


@app.get("/api/v1/data/processed")
def get_processed():
    if _PROCESSED_RECORDS is None:
        return jsonify({"error": "no processed data"}), 400
    return jsonify(_PROCESSED_RECORDS), 200


@app.post("/api/v1/predict")
def predict():
    content_type = (request.content_type or "").split(";")[0].strip().lower()
    if content_type != "application/json":
        return jsonify({"error": "unsupported content type"}), 415

    data = request.get_json(silent=True)
    if not isinstance(data, dict) or "features" not in data:
        return jsonify({"error": "malformed or empty input"}), 400

    feats = data.get("features")
    if not isinstance(feats, list) or len(feats) == 0:
        return jsonify({"error": "malformed or empty input"}), 400

    try:
        arr = np.array(feats, dtype=float)
    except Exception:
        return jsonify({"error": "malformed or empty input"}), 400

    # Deterministic prediction derived from features.
    pred = float(arr.sum())
    return jsonify({"prediction": pred}), 200
