import os
import csv
import json
import glob
import sqlite3
import traceback
from datetime import datetime, timezone
from flask import Flask, render_template, jsonify, send_file, Response

app = Flask(__name__)

DATA_DIR = '/app/workspace/data'
INPUT_DIR = os.path.join(DATA_DIR, 'input')
OUTPUT_FILE = os.path.join(DATA_DIR, 'output.json')
DB_FILE = os.path.join(DATA_DIR, 'etl.db')
LOG_FILE = os.path.join(DATA_DIR, 'etl.log')


def run_etl():
    """Run the ETL pipeline."""
    try:
        # Read all CSV files from input directory
        all_records = []
        csv_files = sorted(glob.glob(os.path.join(INPUT_DIR, '*.csv')))
        for csv_path in csv_files:
            with open(csv_path, 'r', newline='', encoding='utf-8') as f:
                reader = csv.DictReader(f)
                for row in reader:
                    # Ensure all values are strings
                    record = {k: str(v) for k, v in row.items()}
                    all_records.append(record)

        # Write output.json as a flat JSON array
        with open(OUTPUT_FILE, 'w', encoding='utf-8') as f:
            json.dump(all_records, f, indent=2)

        # Insert into SQLite database
        conn = sqlite3.connect(DB_FILE)
        cursor = conn.cursor()
        cursor.execute('CREATE TABLE IF NOT EXISTS records (id INTEGER PRIMARY KEY, data TEXT)')
        for record in all_records:
            cursor.execute('INSERT INTO records (data) VALUES (?)', (json.dumps(record),))
        conn.commit()
        conn.close()

        # Log success
        timestamp = datetime.now(timezone.utc).strftime('%Y-%m-%d %H:%M:%S')
        with open(LOG_FILE, 'a', encoding='utf-8') as f:
            f.write(f'{timestamp} SUCCESS: ETL completed. {len(all_records)} records processed.\n')

        return {'status': 'success', 'records': len(all_records)}

    except Exception as e:
        # Log error
        timestamp = datetime.now(timezone.utc).strftime('%Y-%m-%d %H:%M:%S')
        with open(LOG_FILE, 'a', encoding='utf-8') as f:
            f.write(f'{timestamp} ERROR: {str(e)}\n')
        return {'status': 'error', 'message': str(e)}


@app.route('/')
def index():
    return render_template('index.html')


@app.route('/trigger', methods=['POST'])
def trigger():
    result = run_etl()
    return jsonify(result)


@app.route('/logs')
def logs():
    if os.path.exists(LOG_FILE):
        with open(LOG_FILE, 'r', encoding='utf-8') as f:
            content = f.read()
    else:
        content = ''
    return Response(content, mimetype='text/plain')


@app.route('/download')
def download():
    return send_file(OUTPUT_FILE, as_attachment=True)


if __name__ == '__main__':
    os.makedirs(INPUT_DIR, exist_ok=True)
    app.run(host='0.0.0.0', port=5000)
