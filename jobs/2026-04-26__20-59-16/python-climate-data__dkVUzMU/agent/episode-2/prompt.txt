>     # Ensure keys are strings for robust lookup.

> def _read_and_clean(climate_csv: str, region_map: dict) -> list:
>     cleaned = []
>     with open(climate_csv, "r", encoding="utf-8", newline="") as f:
>         reader = csv.DictReader(f)
>         required_cols = {"region_id", "year", "temperature"}
>         if reader.fieldnames is None or not required_cols.issubset(set(reader.fieldnames)):
>             raise ValueError(f"climate.csv must contain columns {sorted(required_cols)}")
>
>         for row in reader:
>             year = int(row["year"])
>             if year < 2021:
>                 continue  # absolutely no pre-2021 leakage
>
>             region_id = str(row["region_id"])
>             region_name = region_map.get(region_id)
>             if region_name is None:
>                 # Unknown region_id; skip rather than hardcoding/guessing.
>                 continue
>
>             temp = float(row["temperature"])
>             cleaned.append({
>                 "region": str(region_name),
>                 "temperature": float(temp),
>                 "year": int(year),
>             })
>     return cleaned
>
>
> def _compute_trends(cleaned_rows: list) -> dict:
>     sums = defaultdict(float)
>     counts = defaultdict(int)
>     for r in cleaned_rows:
>         region = r["region"]
>         sums[region] += float(r["temperature"])
>         counts[region] += 1
>
>     trends = {}
>     for region, total in sums.items():
>         trends[region] = float(total / counts[region])
>     return trends
>
>
> def _write_json(path: str, obj) -> None:
>     with open(path, "w", encoding="utf-8") as f:
>         json.dump(obj, f, ensure_ascii=False, indent=2)

>
